### Configuration Validation Script

Create `scripts/validate-production-config.sh`:

```bash
#!/bin/bash
set -e

echo "üîç Validating Production Configuration..."

# Check critical environment variables
REQUIRED_VARS=(
  "MONGO_URL"
  "JWT_SECRET"
  "CSRF_SECRET"
  "EMAIL_SERVICE"
  "APP_URL"
  "CORS_ORIGINS"
  "ENVIRONMENT"
)

for var in "${REQUIRED_VARS[@]}"; do
  if [ -z "${!var}" ]; then
    echo "‚ùå ERROR: $var is not set"
    exit 1
  else
    echo "‚úÖ $var is set"
  fi
done

# Check JWT secret strength
if [ ${#JWT_SECRET} -lt 32 ]; then
  echo "‚ùå ERROR: JWT_SECRET must be at least 32 characters"
  exit 1
fi

# Check email service
if [ "$EMAIL_SERVICE" = "mock" ] && [ "$ENVIRONMENT" = "production" ]; then
  echo "‚ùå ERROR: Cannot use mock email service in production"
  exit 1
fi

# Check CORS origins
if [ "$CORS_ORIGINS" = "*" ]; then
  echo "‚ùå ERROR: CORS_ORIGINS cannot be '*' in production"
  exit 1
fi

echo "‚úÖ All configuration checks passed!"
```

Run before deployment:

chmod +x scripts/validate-production-config.sh
./scripts/validate-production-config.sh
